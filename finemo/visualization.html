 30def abbreviate_motif_name(name: str) -> str:
 31    """Convert TF-MoDISco motif names to abbreviated format.
 32
 33    Converts full TF-MoDISco pattern names to shorter, more readable format
 34    for display in plots and reports.
 35
 36    Parameters
 37    ----------
 38    name : str
 39        Full motif name (e.g., 'pos_patterns.pattern_0').
 40
 41    Returns
 42    -------
 43    str
 44        Abbreviated name (e.g., '+/0') or original name if parsing fails.
 45
 46    Examples
 47    --------
 48    >>> abbreviate_motif_name('pos_patterns.pattern_0')
 49    '+/0'
 50    >>> abbreviate_motif_name('neg_patterns.pattern_1')
 51    '-/1'
 52    >>> abbreviate_motif_name('invalid_name')
 53    'invalid_name'
 54    """
 55    try:
 56        group, motif = name.split(".")
 57        if group == "pos_patterns":
 58            group_short = "+"
 59        elif group == "neg_patterns":
 60            group_short = "-"
 61        else:
 62            raise Exception
 63        motif_num = motif.split("_")[1]
 64        return f"{group_short}/{motif_num}"
 65    except Exception:
 66        return name

 69def plot_hit_stat_distributions(
 70    hits_df: pl.LazyFrame, motif_names: List[str], plot_dir: str
 71) -> None:
 72    """Plot distributions of hit statistics for each motif.
 73
 74    Creates separate histogram plots for coefficient, similarity, and importance
 75    score distributions for each motif. Saves plots in both PNG (high-res) and
 76    SVG (vector) formats.
 77
 78    Parameters
 79    ----------
 80    hits_df : pl.LazyFrame
 81        Lazy DataFrame containing hit data with required columns:
 82        - motif_name : str, name of the motif
 83        - hit_coefficient_global : float, global coefficient values
 84        - hit_similarity : float, similarity scores to motif CWM
 85        - hit_importance : float, importance scores from attribution
 86    motif_names : List[str]
 87        List of motif names to generate plots for. Motifs not present
 88        in hits_df will result in empty histograms.
 89    plot_dir : str
 90        Directory path where plots will be saved. Creates subdirectory
 91        'motif_stat_distributions' if it doesn't exist.
 92
 93    Notes
 94    -----
 95    For each motif, creates three separate plots:
 96    - {motif_name}_coefficients.{png,svg} : coefficient distribution
 97    - {motif_name}_similarities.{png,svg} : similarity distribution
 98    - {motif_name}_importances.{png,svg} : importance distribution
 99    """
100    hits_df_collected = hits_df.collect()
101    hits_by_motif = hits_df_collected.partition_by("motif_name", as_dict=True)
102    dummy_df = hits_df_collected.clear()
103
104    motifs_dir = os.path.join(plot_dir, "motif_stat_distributions")
105    os.makedirs(motifs_dir, exist_ok=True)
106    for m in motif_names:
107        hits = hits_by_motif.get((m,), dummy_df)
108        coefficients = hits.get_column("hit_coefficient_global").to_numpy()
109        similarities = hits.get_column("hit_similarity").to_numpy()
110        importances = hits.get_column("hit_importance").to_numpy()
111
112        fig, ax = plt.subplots(figsize=(5, 2))
113
114        # Plot coefficient distribution
115        ax.hist(coefficients, bins=50, density=True)
116
117        output_path_png = os.path.join(motifs_dir, f"{m}_coefficients.png")
118        plt.savefig(output_path_png, dpi=300)
119        output_path_svg = os.path.join(motifs_dir, f"{m}_coefficients.svg")
120        plt.savefig(output_path_svg)
121        plt.close(fig)
122
123        fig, ax = plt.subplots(figsize=(5, 2))
124
125        # Plot similarity distribution
126        ax.hist(similarities, bins=50, density=True)
127
128        output_path_png = os.path.join(motifs_dir, f"{m}_similarities.png")
129        plt.savefig(output_path_png, dpi=300)
130        output_path_svg = os.path.join(motifs_dir, f"{m}_similarities.svg")
131        plt.savefig(output_path_svg)
132        plt.close(fig)
133
134        fig, ax = plt.subplots(figsize=(5, 2))
135
136        # Plot importance distribution
137        ax.hist(importances, bins=50, density=True)
138
139        output_path_png = os.path.join(motifs_dir, f"{m}_importances.png")
140        plt.savefig(output_path_png, dpi=300)
141        output_path_svg = os.path.join(motifs_dir, f"{m}_importances.svg")
142        plt.savefig(output_path_svg)
143        plt.close(fig)

146def plot_hit_peak_distributions(
147    occ_df: pl.DataFrame, motif_names: List[str], plot_dir: str
148) -> None:
149    """Plot distribution of hits per peak for each motif.
150
151    Creates bar plots showing the frequency distribution of hit counts per peak
152    for each motif, plus an overall distribution of total hits per peak.
153
154    Parameters
155    ----------
156    occ_df : pl.DataFrame
157        DataFrame containing motif occurrence counts per peak. Expected to have:
158        - One column per motif name with integer hit counts
159        - 'total' column with sum of all motif hits per peak
160        - Each row represents a peak/genomic region
161    motif_names : List[str]
162        List of motif names corresponding to columns in occ_df.
163    plot_dir : str
164        Directory to save plots. Creates 'motif_hit_distributions' subdirectory.
165
166    Notes
167    -----
168    Generates the following plots:
169    - Individual motif hit distributions: {motif_name}.{png,svg}
170    - Overall hit distribution: total_hit_distribution.{png,svg}
171
172    Bar plots show frequency (proportion) on y-axis and hit count on x-axis.
173    """
174    motifs_dir = os.path.join(plot_dir, "motif_hit_distributions")
175    os.makedirs(motifs_dir, exist_ok=True)
176
177    for m in motif_names:
178        fig, ax = plt.subplots(figsize=(5, 2))
179
180        unique, counts = np.unique(occ_df.get_column(m), return_counts=True)
181        freq = counts / counts.sum()
182        num_bins = np.amax(unique, initial=0) + 1
183        x = np.arange(num_bins)
184        y = np.zeros(num_bins)
185        y[unique] = freq
186        ax.bar(x, y)
187
188        output_path_png = os.path.join(motifs_dir, f"{m}.png")
189        plt.savefig(output_path_png, dpi=300)
190        output_path_svg = os.path.join(motifs_dir, f"{m}.svg")
191        plt.savefig(output_path_svg)
192
193        plt.close(fig)
194
195    fig, ax = plt.subplots(figsize=(8, 4))
196
197    unique, counts = np.unique(occ_df.get_column("total"), return_counts=True)
198    freq = counts / counts.sum()
199    num_bins = np.amax(unique, initial=0) + 1
200    x = np.arange(num_bins)
201    y = np.zeros(num_bins)
202    y[unique] = freq
203    ax.bar(x, y)
204
205    ax.set_xlabel("Total hits per region")
206    ax.set_ylabel("Frequency")
207
208    output_path_png = os.path.join(plot_dir, "total_hit_distribution.png")
209    plt.savefig(output_path_png, dpi=300)
210    output_path_svg = os.path.join(plot_dir, "total_hit_distribution.svg")
211    plt.savefig(output_path_svg, dpi=300)
212
213    plt.close(fig)

216def plot_peak_motif_indicator_heatmap(
217    peak_hit_counts: Int[ndarray, "M M"], motif_names: List[str], output_dir: str
218) -> None:
219    """Plot co-occurrence heatmap showing motif associations across peaks.
220
221    Creates a normalized correlation heatmap showing how frequently pairs of
222    motifs co-occur within the same genomic peaks. Values are normalized by
223    the geometric mean of individual motif frequencies.
224
225    Parameters
226    ----------
227    peak_hit_counts : Int[ndarray, "M M"]
228        Co-occurrence matrix where M = len(motif_names).
229        Entry (i,j) represents the number of peaks containing both motif i and j.
230        Diagonal entries represent total peaks containing each individual motif.
231    motif_names : List[str]
232        List of motif names for axis labels. Order must match matrix dimensions.
233    output_dir : str
234        Directory path where the heatmap plots will be saved.
235
236    Notes
237    -----
238    Saves plots as:
239    - motif_cooocurrence.png : High-resolution raster format
240    - motif_cooocurrence.svg : Vector format
241
242    The heatmap uses correlation normalization: matrix[i,j] / sqrt(matrix[i,i] * matrix[j,j])
243    Colors use the 'Greens' colormap with values typically in [0, 1] range.
244    """
245    cov_norm = 1 / np.sqrt(np.diag(peak_hit_counts))
246    matrix = peak_hit_counts * cov_norm[:, None] * cov_norm[None, :]
247    motif_keys = [abbreviate_motif_name(m) for m in motif_names]
248
249    fig, ax = plt.subplots(figsize=(8, 8), layout="constrained")
250
251    # Plot the heatmap
252    cax = ax.imshow(matrix, interpolation="nearest", aspect="equal", cmap="Greens")
253
254    # Set axes on heatmap
255    ax.set_yticks(np.arange(len(motif_keys)))
256    ax.set_yticklabels(motif_keys)
257    ax.set_xticks(np.arange(len(motif_keys)))
258    ax.set_xticklabels(motif_keys, rotation=90)
259    ax.set_xlabel("Motif i")
260    ax.set_ylabel("Motif j")
261
262    ax.tick_params(axis="both", labelsize=8)
263
264    cbar = fig.colorbar(cax, ax=ax, orientation="vertical", shrink=0.6, aspect=30)
265    cbar.ax.tick_params(labelsize=8)
266
267    output_path_png = os.path.join(output_dir, "motif_cooocurrence.png")
268    plt.savefig(output_path_png, dpi=300)
269    output_path_svg = os.path.join(output_dir, "motif_cooocurrence.svg")
270    plt.savefig(output_path_svg, dpi=300)
271
272    plt.close()

333class LogoGlyph(AbstractPathEffect):
334    """Path effect for creating sequence logo glyphs with normalized dimensions.
335
336    This class creates properly scaled and positioned text glyphs for sequence
337    logos by normalizing character dimensions and applying appropriate transforms.
338
339    Parameters
340    ----------
341    glyph : str
342        Single character to render (e.g., 'A', 'C', 'G', 'T').
343    ref_glyph : str, default 'E'
344        Reference character used for width normalization.
345    font_props : FontProperties, optional
346        Font properties for the glyph rendering.
347    offset : Tuple[float, float], default (0., 0.)
348        Offset for glyph positioning.
349    **kwargs
350        Additional graphics collection parameters.
351    """
352
353    def __init__(
354        self,
355        glyph: str,
356        ref_glyph: str = "E",
357        font_props: Optional[FontProperties] = None,
358        offset: Tuple[float, float] = (0.0, 0.0),
359        **kwargs,
360    ) -> None:
361        super().__init__(offset)
362
363        path_orig = TextPath((0, 0), glyph, size=1, prop=font_props)
364        dims = path_orig.get_extents()
365        ref_dims = TextPath((0, 0), ref_glyph, size=1, prop=font_props).get_extents()
366
367        h_scale = 1 / dims.height
368        ref_width = max(dims.width, ref_dims.width)
369        w_scale = 1 / ref_width
370        w_shift = (1 - dims.width / ref_width) / 2
371        x_shift = -dims.x0
372        y_shift = -dims.y0
373        stretch = (
374            Affine2D()
375            .translate(tx=x_shift, ty=y_shift)
376            .scale(sx=w_scale, sy=h_scale)
377            .translate(tx=w_shift, ty=0)
378        )
379
380        self.path = stretch.transform_path(path_orig)
381
382        #: The dictionary of keywords to update the graphics collection with.
383        self._gc = kwargs
384
385    def draw_path(self, renderer, gc, tpath, affine, rgbFace) -> Any:  # type: ignore[override]
386        """Draw the glyph path using the renderer.
387
388        Parameters
389        ----------
390        renderer : matplotlib renderer
391            The renderer to draw with.
392        gc : GraphicsContext
393            Graphics context for drawing properties.
394        tpath : Path
395            Original text path (unused, using self.path instead).
396        affine : Transform
397            Affine transformation to apply.
398        rgbFace : color
399            Face color for the glyph.
400
401        Returns
402        -------
403        Any
404            Result from renderer.draw_path.
405        """
406        return renderer.draw_path(gc, self.path, affine, rgbFace)

409def plot_logo(
410    ax: Axes,
411    heights: Float[ndarray, "B W"],
412    glyphs: Iterable[str],
413    colors: Optional[Mapping[str, Optional[str]]] = None,
414    font_props: Optional[FontProperties] = None,
415    shade_bounds: Optional[Tuple[int, int]] = None,
416) -> None:
417    """Plot sequence logo from contribution weight matrix.
418
419    Creates a sequence logo visualization where letter heights represent
420    the contribution or information content at each position. Supports
421    both positive and negative contributions with proper stacking.
422
423    Parameters
424    ----------
425    ax : Axes
426        Matplotlib axes object to plot on.
427    heights : Float[ndarray, "B W"]
428        Height matrix where B = len(glyphs) and W = motif width.
429        Entry (i,j) represents the height/contribution of base i at position j.
430        Can contain both positive and negative values.
431    glyphs : Iterable[str]
432        Sequence of base characters corresponding to rows in heights matrix.
433        Typically ['A', 'C', 'G', 'T'] for DNA.
434    colors : Dict[str, str], optional
435        Color mapping for each base. Keys should match glyphs.
436        If None, all bases will use default matplotlib colors.
437    font_props : FontProperties, optional
438        Font properties for letter rendering. If None, uses default font.
439    shade_bounds : Tuple[int, int], optional
440        (start, end) position indices to shade in background.
441        Useful for highlighting core motif regions.
442
443    Notes
444    -----
445    Positive and negative contributions are handled separately:
446    - Positive values are stacked above zero line in order of descending absolute value
447    - Negative values are stacked below zero line in order of descending absolute value
448    - A horizontal line is drawn at y=0 for reference
449
450    The resulting plot has:
451    - X-axis: Position in motif (0-indexed)
452    - Y-axis: Contribution magnitude
453    - Bar width: 0.95 (small gaps between positions)
454    """
455    if colors is None:
456        colors = {g: None for g in glyphs}
457
458    ax.margins(x=0, y=0)
459
460    pos_values = np.clip(heights, 0, None)
461    neg_values = np.clip(heights, None, 0)
462    pos_order = np.argsort(pos_values, axis=0)
463    neg_order = np.argsort(neg_values, axis=0)[::-1, :]
464    pos_reorder = np.argsort(pos_order, axis=0)
465    neg_reorder = np.argsort(neg_order, axis=0)
466    pos_offsets = np.take_along_axis(
467        np.cumsum(np.take_along_axis(pos_values, pos_order, axis=0), axis=0),
468        pos_reorder,
469        axis=0,
470    )
471    neg_offsets = np.take_along_axis(
472        np.cumsum(np.take_along_axis(neg_values, neg_order, axis=0), axis=0),
473        neg_reorder,
474        axis=0,
475    )
476    bottoms = pos_offsets + neg_offsets - heights
477
478    x = np.arange(heights.shape[1])
479
480    for glyph, height, bottom in zip(glyphs, heights, bottoms):
481        ax.bar(
482            x,
483            height,
484            0.95,
485            bottom=bottom,
486            path_effects=[LogoGlyph(glyph, font_props=font_props)],
487            color=colors[glyph],
488        )
489
490    if shade_bounds is not None:
491        start, end = shade_bounds
492        ax.axvspan(start - 0.5, end - 0.5, color="0.9", zorder=-1)
493
494    ax.axhline(zorder=-1, linewidth=0.5, color="black")
495

502def plot_cwms(
503    cwms: Dict[str, Dict[str, Float[ndarray, "4 W"]]],
504    trim_bounds: Dict[str, Dict[str, Tuple[int, int]]],

564            for name, spine in ax.spines.items():

566
567            output_path_png = os.path.join(motif_dir, f"{cwm_type}.png")

569            output_path_svg = os.path.join(motif_dir, f"{cwm_type}.svg")
570            plt.savefig(output_path_svg)
571
572            plt.close(fig)

575def plot_hit_vs_seqlet_counts(
576    recall_data: Dict[str, Dict[str, Union[int, float]]], output_dir: str
577) -> None:
578    """Plot scatter plot comparing hit counts to seqlet counts per motif.
579
580    Creates a log-log scatter plot showing the relationship between the number
581    of hits called by Fi-NeMo and the number of seqlets identified by TF-MoDISco
582    for each motif. Includes diagonal reference line and motif annotations.
583
584    Parameters
585    ----------
586    recall_data : Dict[str, Dict[str, Union[int, float]]]
587        Dictionary with motif names as keys and metrics dictionaries as values.
588        Each metrics dictionary must contain:
589        - 'num_hits_total' : int, total number of hits for the motif
590        - 'num_seqlets' : int, total number of seqlets for the motif
591    output_dir : str
592        Directory path where the scatter plot will be saved.
593
594    Notes
595    -----
596    Saves plots as:
597    - hit_vs_seqlet_counts.png : High-resolution raster format
598    - hit_vs_seqlet_counts.svg : Vector format
599
600    Plot features:
601    - Log-log scale on both axes
602    - Diagonal reference line (y = x) as dashed line
603    - Points annotated with abbreviated motif names
604    """
605    x = []
606    y = []
607    m = []
608    for k, v in recall_data.items():
609        x.append(v["num_hits_total"])
610        y.append(v["num_seqlets"])
611        m.append(k)
612

455    """

458

460
461    pos_values = np.clip(heights, 0, None)
462    neg_values = np.clip(heights, None, 0)
463    pos_order = np.argsort(pos_values, axis=0)
464    neg_order = np.argsort(neg_values, axis=0)[::-1, :]
465    pos_reorder = np.argsort(pos_order, axis=0)
466    neg_reorder = np.argsort(neg_order, axis=0)
467    pos_offsets = np.take_along_axis(
468        np.cumsum(np.take_along_axis(pos_values, pos_order, axis=0), axis=0),

470        axis=0,
471    )
472    neg_offsets = np.take_along_axis(

474        neg_reorder,
475        axis=0,
476    )

478
479    x = np.arange(heights.shape[1])
